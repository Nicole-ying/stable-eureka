```python
def compute_reward(obs, action, next_obs, done, info):
    # Unpack observation components (normalized)
    x, y = obs[0], obs[1]
    vx, vy = obs[2], obs[3]
    angle = obs[4]
    ang_vel = obs[5]
    left_contact = obs[6]
    right_contact = obs[7]

    # ---------- 1. velocity_penalty ----------
    # Penalize high speeds, especially vertical velocity.
    # Scale: typical |vy| up to ~0.5, |vx| up to ~0.5, so penalty range ~0-3.
    vel_penalty = -2.0 * (abs(vy) + 0.3 * abs(vx))

    # ---------- 2. angle_penalty ----------
    # Penalize tilt and spin.
    # Scale: |angle| up to ~0.5 rad, |ang_vel| up to ~0.5, so penalty range ~0-3.
    angle_penalty = -3.0 * (abs(angle) + 0.3 * abs(ang_vel))

    # ---------- 3. distance_reward ----------
    # Negative Euclidean distance from pad (0,0), scaled moderately.
    # Typical distance range: 0 to ~1.5, so reward range ~0 to -6.
    dist = np.sqrt(x**2 + y**2)
    distance_reward = -4.0 * dist

    # ---------- 4. fuel_efficiency_penalty ----------
    # Penalize engine usage: main engine (action==2) and side engines (action==1 or 3).
    fuel_penalty = 0.0
    if action == 2:          # main engine
        fuel_penalty = -2.0
    elif action in [1, 3]:   # side engines
        fuel_penalty = -1.0

    # ---------- 5. ground_contact_bonus ----------
    # Provide bonus when both legs are on ground and lander is near pad.
    # Relaxed conditions: |x| < 0.3, y < 0.2 (near ground and near pad center).
    near_pad = (abs(x) < 0.3) and (y < 0.2)
    both_legs = (left_contact > 0.5) and (right_contact > 0.5)
    ground_contact_bonus = 5.0 if (both_legs and near_pad) else 0.0

    # ---------- 6. terminal ----------
    # Apply only when episode ends (done == True).
    # Successful landing: both legs on ground, near pad, low speed, upright.
    terminal_reward = 0.0
    if done:
        success = (
            both_legs and near_pad and
            abs(vy) < 0.1 and abs(vx) < 0.1 and abs(angle) < 0.1
        )
        if success:
            terminal_reward = 100.0
        else:
            # Moderate penalty for failure (crash or out-of-bounds)
            terminal_reward = -50.0

    # Sum all components
    total_reward = (
        vel_penalty +
        angle_penalty +
        distance_reward +
        fuel_penalty +
        ground_contact_bonus +
        terminal_reward
    )

    # Clamp total reward to safe bounds
    total_reward = np.clip(total_reward, -1000.0, 1000.0)

    # Build components dictionary (all required IDs)
    components_dict = {
        "velocity_penalty": vel_penalty,
        "angle_penalty": angle_penalty,
        "distance_reward": distance_reward,
        "fuel_efficiency_penalty": fuel_penalty,
        "ground_contact_bonus": ground_contact_bonus,
        "terminal": terminal_reward
    }

    return float(total_reward), components_dict
```

RATIONALE:
This reward function addresses the failure modes observed in Generation 0. Key changes:
- Reduced shaping penalty magnitudes (velocity_penalty -2 vs -3, angle_penalty -3 vs -4) to avoid overwhelming positive signals.
- Increased fuel penalties (-2 for main, -1 for side) to discourage wasteful engine firing while still allowing exploration.
- Relaxed ground_contact_bonus conditions (|x|<0.3, y<0.2) to make the bonus more achievable.
- Reduced terminal failure penalty from -100 to -50 to prevent it from dominating the total reward.
- Used obs (not next_obs) for leg contacts and state, as the previous generation showed no clear benefit from next_obs.
- Maintained the same success criteria for terminal bonus (+100) to incentivize proper landings.
- All component IDs match the required schema.
